# Remove dangling images
docker images -qf dangling=true | xargs docker rmi

# Remove dangling volumes
docker volume ls -qf dangling=true | xargs -r docker volume rm

# docker image remove based on string matching
docker images | grep "dkr.ecr.us-east-" | awk '{print $3}' | xargs docker rmi

# docker image remove based on matching tags: 
docker images | grep "dkr.ecr.us-east-" | awk '{print $1 ":" $2}' | xargs docker rmi

# docker image remove based on matching tags: 
docker images | grep "cdkasset-" | awk '{print $1 ":" $2}' | xargs docker rmi

# Remove unused images
docker image prune -a

# Remove build cache (largest gain)
docker builder prune -a

# Remove all following docker compose file configuration
docker compose down -v --rmi all --remove-orphans
